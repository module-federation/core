# React Bridge（for React v18）

`@module-federation/bridge-react` 提供了用于 react v18 应用的 `bridge` 工具函数，其提供的 `createBridgeComponent` 可用于导出应用级别模块，`createRemoteComponent` 用于加载应用级别模块。[Demo](https://github.com/module-federation/core/blob/6756c9142d0bf41262a26fe30e3c39d7957cfa58/apps/router-demo)



### 安装

import { PackageManagerTabs } from '@theme';

<PackageManagerTabs

  }}
/>


### 示例

> Remote

```tsx
// ./src/export-app.tsx
import App from './src/App.tsx';
import { createBridgeComponent } from '@module-federation/bridge-react';

export default createBridgeComponent({
  rootComponent: App
});
```

```ts
// rsbuild.config.ts
export default defineConfig({
  source: {
    alias: {
      // 设置 react-router-dom 到 proxy 的 router 中，方便
      'react-router-dom$': path.resolve(
        __dirname,
        'node_modules/@module-federation/bridge-react/dist/router.es.js',
      ),
    },
  },
  tools: {
    rspack: (config, { appendPlugins }) => {
      appendPlugins([
        new ModuleFederationPlugin({
          name: 'remote1',
          exposes: {
            './export-app': './src/export-app.tsx',
          },
          shared: ['react', 'react-dom'],
        }),
      ]);
    },
  },
});
```

### 方法

#### createRemoteComponent

```tsx
function createRemoteComponent<T, `E extends keyof T`>(
  // 加载远程应用的函数，例如：loadRemote('remote1/export-app')、import('remote1/export-app')
  lazyComponent: () => Promise<T>,
  options?: {
    // 默认为 default，用于指定模块的 export
    export?: E;
  }
): (props: {
    basename?: ProviderParams['basename'];
    memoryRoute?: { entryPath: string };
    fallback: ReactNode;
} & RawComponentType) => React.JSX.Element;
```

* `lazyComponent`
  * type: `() => Promise<Module>`
  * 作用: 用于加载远程模块的函数，例如：`loadRemote('remote1/export-app')`、`import('remote1/export-app')`

```tsx
const Remote1App = createRemoteComponent(() => loadRemote('remote1/export-app'));
const Remote2App = createRemoteComponent(() => import('remote2/export-app'));
```

* `options`
  * `export`
    * type: `string`
    * 作用: 可以指定模块的 export
```tsx
// remote
export const provider = createBridgeComponent({
  rootComponent: App
});

// host
const Remote1App = createRemoteComponent(() => loadRemote('remote1/export-app'), {
  export: 'provider'
});
```

* ReturnType
  * type: `(props: PropsInfo)=> React.JSX.Element`
  * 作用: 用于渲染远程模块组件

```tsx
const Remote1App = createRemoteComponent(() => loadRemote('remote1/export-app'));

function App() {
  return (<BrowserRouter basename="/">
    <Routes>
      <Route
        path="/remote1/*"
        Component={() => <Remote1App fallback={<div>loading</div>} />}
      />
    </Routes>
  </BrowserRouter>)
}
```

```tsx
const Remote1App = createRemoteComponent(() => loadRemote('remote1/export-app'));

function App() {
  return (<BrowserRouter basename="/">
    <Routes>
      <Route
        path="/remote1/*"
        {/* 通过 memoryRoute 来讲子应用路由控制为 memoryRouter，将不会直接将 url 展示在浏览器地址上  */}
        Component={() => <Remote1App fallback={<div>loading</div>} memoryRoute={{ entryPath: '/detail' }}/>}
      />
    </Routes>
  </BrowserRouter>)
}
```

#### createBridgeComponent

```tsx
function createBridgeComponent<T>(bridgeInfo: {
  rootComponent: React.ComponentType<T>;
}): () => {
  render(info: {
    name?: string;
    basename?: string;
    memoryRoute?: {
      entryPath: string;
    };
    dom?: HTMLElement;
}): void;
  destroy(info: { dom: HTMLElement}): void;
}
```

* `bridgeInfo`
  * type: `{ rootComponent: React.ComponentType<T>; }`
  * 作用: 用于传递根组件
* ReturnType
  * type: `() => { render: (info: RenderFnParams) => void; destroy: (info: { dom: HTMLElement}) => void; }`

```tsx
// ./src/export-app.tsx
import React from 'react';
import App from './src/App.tsx';
import { createRemoteComponent } from '@module-federation/bridge-react';

export default createBridgeComponent({
  rootComponent: App
});
```

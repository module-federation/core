This example reflects the current `nextjs-mf` contract:

- use `next-rspack`
- pass standard Module Federation options to `@module-federation/nextjs-mf`
- use `@module-federation/nextjs-mf/server` for Pages SSR asset flushing
- explicitly expose any stitched Pages routes you import from another app
- do **not** use `extraOptions`, `./utils`, `pages-map`, or `revalidate()`

## Install

```bash
pnpm add next-rspack @rspack/core @module-federation/nextjs-mf
```

## Configure a Host

```js title="next.config.js"
const withRspack = require('next-rspack');
const NextFederationPlugin = require('@module-federation/nextjs-mf');

/** @type {import('next').NextConfig} */
const nextConfig = {
  webpack(config, options) {
    config.plugins.push(
      new NextFederationPlugin({
        name: 'home',
        filename: 'static/chunks/remoteEntry.js',
        remotes: {
          shop: `shop@http://localhost:3001/_next/static/${
            options.isServer ? 'ssr' : 'chunks'
          }/remoteEntry.js`,
        },
        exposes: {
          './menu': './components/menu',
          './pages/index': './pages/index',
        },
      }),
    );

    return config;
  },
};

module.exports = withRspack(nextConfig);
```

## Configure a Remote

```js title="next.config.js"
const withRspack = require('next-rspack');
const NextFederationPlugin = require('@module-federation/nextjs-mf');

/** @type {import('next').NextConfig} */
const nextConfig = {
  webpack(config) {
    config.plugins.push(
      new NextFederationPlugin({
        name: 'shop',
        filename: 'static/chunks/remoteEntry.js',
        exposes: {
          './menu': './components/menu',
          './pages/shop/index': './pages/shop/index',
        },
      }),
    );

    return config;
  },
};

module.exports = withRspack(nextConfig);
```

## Add the Pages SSR Helpers

```jsx title="pages/_document.js"
import Document, { Html, Head, Main, NextScript } from 'next/document';
import {
  FlushedChunks,
  flushChunks,
  withFederatedRequest,
} from '@module-federation/nextjs-mf/server';

class MyDocument extends Document {
  static async getInitialProps(ctx) {
    const { initialProps, chunks } = await withFederatedRequest(async () => {
      const initialProps = await Document.getInitialProps(ctx);
      const chunks = await flushChunks();

      return { initialProps, chunks };
    });

    return {
      ...initialProps,
      chunks,
    };
  }

  render() {
    return (
      <Html>
        <Head>
          <FlushedChunks chunks={this.props.chunks} />
        </Head>
        <body>
          <Main />
          <NextScript />
        </body>
      </Html>
    );
  }
}

export default MyDocument;
```

## Import the Remote

```jsx title="pages/index.js"
import React, { lazy } from 'react';

const Menu = lazy(() => import('shop/menu'));

export default function Page() {
  return <Menu />;
}
```

## Current Limitations

- No `extraOptions`
- No `enableImageLoaderFix` / `enableUrlLoaderFix`
- No `./utils`
- No `pages-map`
- No `revalidate()` export from `@module-federation/nextjs-mf/server`
- No App Router support claim in wave 1
